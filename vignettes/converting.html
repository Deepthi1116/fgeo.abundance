From array to data frame

Example grouping by two variables.

# USING `abundance_n()`

abundance_n <- function(x, group_by = c("quadrat", "sp"), only_alive = TRUE) {
    if (only_alive) {
      valid.status <- "A"
    } else {
      valid.status <- unique(x$status)
    }
    table(x[x$status %in% valid.status, group_by])
}

From data frame to array

plyr can convert from class data.frame to class array.

df <- abundance(stem, group_by = c("sp", "status"), only_alive = FALSE)
head(df)
#>       sp status  n
#> 1 acaldi      A  5
#> 2 acaldi      D 11
#> 3 acaldi      G 11
#> 4 aegipa      D  1
#> 5 alchco      D  1
#> 6 alchco      G 10
a <- daply(df, .(sp, status), nrow)
head(a)
#>         status
#> sp        A D  G
#>   acaldi  1 1  1
#>   aegipa NA 1 NA
#>   alchco NA 1  1
#>   alibed  1 1 NA
#>   allops NA 1 NA
#>   alsebl  1 1  1
is.array(a)
#> [1] TRUE
